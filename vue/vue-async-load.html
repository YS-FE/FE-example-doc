vue组件异步加载

组件异步加载

注册组件

//全局组件
Vue.component('async-example', function (resolve, reject) {
    axios.get('/xxx').then(data => {
     //data是个对象 {template: `<h1>sgsgsg</h1>`}
     resolve(data); 
    })
 })

 //局部组件
 new Vue({
     ....,
     components: {
         xxx: function(resolve, reject){
               axios.get('/xxx').then(data => {
            //data是个对象 {template: `<h1>sgsgsg</h1>`}

             resolve(data); //resolve(Vue.extend(data)也可以
            })
         }
     }
 })
